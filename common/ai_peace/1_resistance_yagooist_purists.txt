#  This ai_peace file is designed for pro_yagoo_conservatism (democratic), pro_yagoo_progressive, pro_resistance_revolutionary (communism) and 
#  mixed_ideology_democracy
#  Unique Resistance and Yagooist ideologies (conservative, progressive, revolutionary, mixed ideology democracy) would aim to change government type and liberate
#  when it makes sense for them. They will only annex cores. Puppets would be rare 
# The resistance one is in another file with only minor differences

hloi_democracy_yagooist_peace = {
	enable = {
		OR = {
		has_government = democratic
		has_government = pro_yagoo_progressive
		has_government = mixed_ideology_democracy
		}
	}

	annex_randomness = 0 
	liberate_randomness = 50 
	puppet_randomness = 0 
	take_states_randomness = 0 
	force_government_randomness = 100 
	
	# ROOT = Taker, FROM = Giver
	annex = {
		factor = 25
		
		modifier = {
			factor = 0
			OR = {
				NOT = {
					has_annex_war_goal = FROM
				}
				NOT = {
					is_owner_neighbor_of = FROM
				}
			}
			FROM = {
				any_owned_state = { 
					NOT = {
						OR = { 
							is_claimed_by = ROOT
							is_core_of = ROOT
							has_claimed_state_in_peace_conference = ROOT
						}
					}
				}
			}
		}
		#modifier = {
		#	factor = 0
		#	is_puppet = yes
		#}		
	}

	# ROOT = Taker, FROM = Liberated
	liberate = { #Liberate might be buggy and might require country-specific workarounds...
		factor = 800 

		modifier = {
			factor = 0

			NOT = {
				ai_liberate_desire = {
					target = FROM
					value > 0
				}
			}		
			NOT = {
				FROM = {
					has_government = democratic
					has_government = pro_yagoo_progressive
					has_government = mixed_ideology_democracy
				}			
			}
		}
	}

	# ROOT = Taker, FROM = Giver
	puppet = {
		factor = 100
		modifier = {
			factor = 0
			NOT = { is_owner_neighbor_of = FROM }
			NOT = {
				FROM = {
					any_neighbor_country = {
						is_puppet_of = ROOT
					}
				}
			}
			NOT = {
				FROM = {
					any_owned_state = {
						any_neighbor_state = {
							OR = {
								owner = { tag = ROOT }
								owner = { is_puppet_of = ROOT }
								has_claimed_state_in_peace_conference = ROOT
							}
						}
					}
				}
			}
			OR = {
				FROM = { capital_scope = { is_on_continent = europe } }
				NOT = {
					FROM = {
						any_owned_state = { 
							OR = {
								is_coastal = yes
								is_island_state = yes
							}
						}
					}
				}
				NOT = { any_owned_state = { OR = { is_coastal = yes is_island_state = yes } } }
			}
		}
		modifier = {
			factor = 100 #Aqua, Pekora and Coco are naughty >:((
			FROM = {
				OR = {	
					tag = AQA
					tag = PEK
					tag = COC
				}
			}
		}		
		modifier = {
			factor = 0.5 #Strip colonies/liberate first - important vs GER and JAP
			FROM = {
				any_owned_state = { NOT = { is_core_of = FROM } }
			}
		}
		modifier = { 
			factor = 0
			OR = {
				AND = {
					#Is NOT a major
					NOT = { num_of_factories > 100 }
				}
				is_puppet = yes
			}
		}
		modifier = {
			factor = 0
			NOT = { is_owner_neighbor_of = FROM }
			#Is NOT a major
			NOT = { num_of_factories > 100 }
		}
		modifier = { #Don't puppet nations in civil war... this is too buggy.
			factor = 0
			FROM = { has_civil_war = yes }
		}
	}

	# ROOT = Taker, FROM = Giver
	puppet_all = {
		base = 0 
	}
	
	# ROOT = STATE, FROM = Taker, FROM.FROM = Giver
	puppet_state = {
		base = 0 

	}
	#Democracy ftw!
	force_government = {
		base = 900
		
		modifier = {
			factor = 0
			#naughty peko,coco,aqua
			FROM = {
				NOT = {
					tag = AQA
					tag = COC
					tag = PEK
				}
			}
		}
	}
}
