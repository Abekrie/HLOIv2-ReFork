#  This ai_peace file is designed for new nationalism (fascism), left_wing_revolutionary, democratic revitalisation, reactionary_monarchy
#  Democratic Antis would liberate non-Hololive countries wherever possible, revolutionary antis would annex, puppet and liberate to equal degrees
#  Nationalist and Monarchist antis would puppet and annex 

democratic_antis_peace = {
	enable = {
		has_government = democratic_revitalisation
	}
	
	annex_randomness = 0 
	liberate_randomness = 300 
	puppet_randomness = 0
	take_states_randomness = 0 
	force_government_randomness = 10 
	
	annex = {
		factor = 10
	}
	
	take_states = {
		factor = 250
		
		modifier = {
			factor = 0
			OR = {
				NOT = {
					has_annex_war_goal = FROM
				}
				NOT = {
					is_owner_neighbor_of = FROM
				}
			}
			FROM = {
				any_owned_state = { 
					NOT = {
						OR = { 
							is_claimed_by = ROOT
							is_core_of = ROOT
							has_claimed_state_in_peace_conference = ROOT
						}
					}
				}
			}
		}
		
		modifier = {
			factor = 0
			FROM = {
				has_government = democratic_revitalisation
			}
		}
		modifier = {
			factor = 2
			FROM = {
				AND = {
					is_in_home_area = yes
					is_hololive_idelogy = yes
				}
			}
		}
		
		modifier = {
			factor = 100
			FROM = {
				any_owned_state = { is_core_of = ROOT }
			}
		}
	}
	# ROOT = Taker, FROM = Liberated
	liberate = { #Liberate might be buggy and might require country-specific workarounds...
		factor = 900 

		modifier = {
			factor = 0

			NOT = {
				ai_liberate_desire = {
					target = FROM
					value > 0
				}
			}		
			NOT = {
				FROM = {
					has_government = democratic_revitalisation
					has_government = mixed_ideology_democracy
				}			
			}
		}
				
		modifier = {
			factor = 700 #Strip colonies/liberate first - important vs GER and JAP
			FROM = {
				any_owned_state = { NOT = { is_core_of = FROM } }
			}
		}
	}	
	puppet = {
		factor = 150
		
		modifier = {
			factor = 10 #Strip colonies/liberate first - important vs GER and JAP
			FROM = {
				any_owned_state = { NOT = { is_core_of = FROM } }
			}
		}

	}
	force_government = {
		factor = 100
		
		modifier = {
		factor = 1000
			FROM = {
				OR = {
					has_government= fascism
					has_government= left_wing_revolutionary
					has_government= reactionary_monarchy
				}
			}
		}
	}
}
