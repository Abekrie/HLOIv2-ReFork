construction_war_annex_country_pek = {
	if = {
		limit = {
			THIS = {
				check_variable = {
					var = THIS.hololive_investment_pek
					value = 15
					compare = greater_than_or_equals
				}
			}
		}
		PEK = {
			annex_country = {
				target = PREV
				transfer_troops = yes
			}
		}
	}
}

construction_war_pek_sub = {
	if = {
		limit = {
			THIS = {
				check_variable = {
					var = THIS.hololive_investment_pek
					value = 10
					compare = greater_than_or_equals
				}
				check_variable = {
					var = THIS.hololive_investment_sub
					value = 10
					compare = greater_than_or_equals
				}
			}
		}
		add_to_array = {
			array = global.construction_war_escalator
			value = THIS.id
		}
		PEK = {
			country_event = investment.2
		}
		SUB = {
			country_event = investment.2
		}
	}
}

construction_war_battle_royal = {
	if = {
		limit = {
			check_variable = {
				var = THIS.hololive_investment_aqa
				value = 10
				compare = greater_than_or_equals
			}
			check_variable = {
				var = THIS.hololive_investment_pek
				value = 10
				compare = greater_than_or_equals
			}
			check_variable = {
				var = THIS.hololive_investment_sub
				value = 10
				compare = greater_than_or_equals
			}
		}
		PEK = {
			declare_war_on = {
				target = AQA
			}
			declare_war_on = {
				target = SUB
			}
		}
		AQA = {
			declare_war_on = {
				target = SUB
			}
		}
		set_global_flag = Construction_wars_have_begun
		news_event = investment.3
	}
}
