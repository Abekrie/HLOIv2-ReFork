WTS_south_england_frame_effect = {
	if = {
		limit = {
			WTS_south_england_government_frame_trigger = yes
		}
		set_variable = {
			var = WTS.counter_insurgency_south_england_frame
			value = 1
		}
		else_if = {
			limit = {
				WTS_south_england_antis_frame_trigger = yes
			}
			set_variable = {
				var = WTS.counter_insurgency_south_england_frame
				value = 2
			}
		}
		else_if = {
			limit = {
				WTS_south_england_irish_frame_trigger = yes
			}
			set_variable = {
				var = WTS.counter_insurgency_south_england_frame
				value = 3
			}
		}
		else_if = {
			limit = {
				WTS_south_england_teammates_frame_trigger = yes
			}
			set_variable = {
				var = WTS.counter_insurgency_south_england_frame
				value = 4
			}
		}
		else_if = {
			limit = {
				WTS_south_england_welsh_frame_trigger = yes
			}
			set_variable = {
				var = WTS.counter_insurgency_south_england_frame
				value = 5
			}
		}
		else = {
			set_variable = {
				var = WTS.counter_insurgency_south_england_frame
				value = 6
			}
		}
	}
}
