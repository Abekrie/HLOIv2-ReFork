seiso_refresh = {
		if = {	
			limit = {
				check_variable = {
					var = ROOT.seiso_alignment
					value = 1
				}
			}
			seiso_clear = yes
			add_ideas = seiso_value_1
		}
		else_if = {	
			limit = {
				check_variable = {
					var = ROOT.seiso_alignment
					value = 2
				}
			}
			seiso_clear = yes
			add_ideas = seiso_value_2
		}
		else_if = {	
			limit = {
				check_variable = {
					var = ROOT.seiso_alignment
					value = 3
				}
			}
			seiso_clear = yes
			add_ideas = seiso_value_3
		}
		else_if = {	
			limit = {
				check_variable = {
					var = ROOT.seiso_alignment
					value = 4
				}
			}
			seiso_clear = yes
			add_ideas = seiso_value_4
		}
		else_if = {	
			limit = {
				check_variable = {
					var = ROOT.seiso_alignment
					value = 5
				}
			}
			seiso_clear = yes
			add_ideas = seiso_value_5
		}
}

capitalism_plus_bar_scripted_effect = {
			if = {
				limit = {
					has_elections = yes
				}
				add_to_variable = {
					var = ROOT.capitalism_alignment
					value = 1
				}
				set_country_flag = {
					flag = democratic_government_change_cooldown
					days = 120
					value = 1
				}
			}
		else_if = {
			add_to_variable = {
				var = ROOT.capitalism_alignment
				value = 1
			}
			set_country_flag = {
				flag = autocratic_government_change_cooldown
				days = 60
				value = 1
			}
		}						
}
