on_actions = {
	on_startup = {
		effect = {
			#HIGHER SCORE IS MORE LOYALTY TOWARDS FUBUKI, LESS IS MORE LOYALTY TOWARDS WFA
			WFA = {
				set_variable = { WFA.loyalty_meter = 40 }
				clamp_variable = { var = WFA.loyalty_meter min = 0 max = 100 }
			}
			FBK = {

				# Relationship value pre modifier - Use this for increase / decrease relationship events  
				set_variable = { FBK.WFA_relationship_flat_var = 50 }	

				# Current Relationship value - Calculated and Updated everyday - Use this for comparing value events
				set_variable = { FBK.WFA_relationship_var = 50 }  		
				
				# Army Ratio modifier
				set_variable = { FBK.army_balance_modifier = 0 } 		
                
                clamp_variable = { var = FBK.WFA_relationship_var min = 0 max = 100 }
				clamp_variable = { var = FBK.WFA_relationship_flat_var min = 0 max = 100 }
				
				FBK_army_variables = yes	
			}
		}
	}
	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = FBK }
					FROM = { NOT = { tag = FBK } }
				}
				every_country = {
					limit = {
						has_country_flag = promised_fbk_war_aid
					}
					country_event = {
						id = fubuki.18
						days = 1
					}
					clr_country_flag = promised_fbk_war_aid
				}
			}
			else_if = {
				limit = {
					ROOT = { NOT = { tag = FBK } }
					FROM = { tag = FBK  }
				}
				every_country = {
					limit = {
						has_country_flag = promised_fbk_war_aid
					}
					country_event = {
						id = fubuki.18
						days = 1
					}
					clr_country_flag = promised_fbk_war_aid
				}
			}
		}
	}
	on_daily_FBK ={
		effect = {
			FBK_army_variables = yes
		}
	}
}
