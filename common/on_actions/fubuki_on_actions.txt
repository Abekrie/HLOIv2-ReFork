on_actions = {
	on_startup = {
		effect = {
			#HIGHER SCORE IS MORE LOYALTY TOWARDS FUBUKI, LESS IS MORE LOYALTY TOWARDS WFA
			WFA = {
				set_variable = { WFA.loyalty_meter = 40 }
				clamp_variable = { var = WFA.loyalty_meter min = 0 max = 100 }
			}
			FBK = {
				 
				set_variable = { FBK.WFA_relationship_flat_var = 0 }	# Relationship value pre modifier - Use this for events
				set_variable = { FBK.WFA_relationship_var = 0 }  		# Current Relationship value - Calculated and Updated everyday 
				set_variable = { FBK.army_balance_modifier = 0 } 		# Army Ratio modifier
                
                clamp_variable = { var = FBK.WFA_relationship_var min = 0 max = 100 }
				clamp_variable = { var = FBK.army_balance_modifier min = 0 max = 100 }
			
		}
	}
	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = FBK }
					FROM = { NOT = { tag = FBK } }
				}
				every_country = {
					limit = {
						has_country_flag = promised_fbk_war_aid
					}
					country_event = {
						id = fubuki.18
						days = 1
					}
					clr_country_flag = promised_fbk_war_aid
				}
			}
			else_if = {
				limit = {
					ROOT = { NOT = { tag = FBK } }
					FROM = { tag = FBK  }
				}
				every_country = {
					limit = {
						has_country_flag = promised_fbk_war_aid
					}
					country_event = {
						id = fubuki.18
						days = 1
					}
					clr_country_flag = promised_fbk_war_aid
				}
			}
		}
	}
}
on_daily_FBK ={
	effect = {
		set_temp_variable = { army_balance_modifier_value = FBK.WFA_relationship_flat_var }
		set_temp_variable = { relationship_original_value = FBK.WFA_relationship_flat_var }

		multiply_temp_variable = {army_balance_modifier_value = FBK.army_balance_modifier }		
		add_to_temp_variable = { relationship_original_value = army_balance_modifier_value }

		set_variable = { FBK.WFA_relationship_var = relationship_original_value }	
	}
}
}
