on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = {
					NOT = {
						has_country_flag = government_alignment_set
					}
				}
				set_variable = {
					var = THIS.seiso_alignment
					value = 3
				}
				clamp_variable = {
					var = THIS.seiso_alignment
					min = 1
					max = 5
				}
				set_variable = {
					var = THIS.reactionary_alignment
					value = 3
				}
				clamp_variable = {
					var = THIS.reactionary_alignment
					min = 1
					max = 5
				}
				set_variable = {
					var = THIS.authoritarianism_alignment
					value = 3
				}
				clamp_variable = {
					var = THIS.authoritarianism_alignment
					min = 1
					max = 5
				}
				set_variable = {
					var = THIS.capitalism_alignment
					value = 3
				}
				clamp_variable = {
					var = THIS.capitalism_alignment
					min = 1
					max = 5
				}
				set_variable = {
					var = THIS.isolationism_alignment
					value = 3
				}
				clamp_variable = {
					var = THIS.isolationism_alignment
					min = 1
					max = 5
				}
				government_alignment_initiate = yes
			}
		}
	}
}
