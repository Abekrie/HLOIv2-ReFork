CAT_form_china = {
	GEN_form_china = {
		icon = GFX_decision_icon_default
		cost = 150
		fire_only_once = yes
		available = {
			has_political_power > 150
			hidden_trigger = {
				OR = {
					tag = GAW
					tag = NIN
					tag = NEN
					tag = XSM
					tag = LUN
					tag = TOW
					tag = WAT
					tag = YOG
					tag = CIV
					tag = ART
					tag = RSA
					tag = CPG
					tag = DOR
				}
				can_form_china = yes
			}
			custom_trigger_tooltip = {
				tooltip = can_form_china_trigger_tooltip
			}
		}
		visible = {
			can_see_form_china = yes
		}
		complete_effect = {
			custom_effect_tooltip = can_form_china_effect_tooltip
			set_temp_variable = {
				var = get_tag_for_cosmetic
				value = ROOT.id
			}
			meta_effect = {
				text = {
					set_cosmetic_tag = [get_tag_for_cosmetic]_china
				}
				get_tag_for_cosmetic = "[?get_tag_for_cosmetic]"
			}
			hidden_effect = {
				form_china = yes
			}
		}
	}
	GEN_create_puppet_china = {
		icon = GFX_decision_icon_default
		cost = 150
		fire_only_once = yes
		available = {
			hidden_trigger = {
				NOT = {

					tag = DOR
				}
				can_form_china = yes
			}
			custom_trigger_tooltip = {
				tooltip = can_form_china_trigger_tooltip
			}
		}
		visible = {
			can_see_form_china = yes
		}
		complete_effect = {
			custom_effect_tooltip = can_form_china_effect_tooltip
			hidden_effect = {
				make_puppet_china = yes
