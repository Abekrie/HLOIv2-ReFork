noe_hololive_union = {
				join_hololive_union = {
				icon = GFX_decision_icon_default
		
		complete_effect = {
				every_country = {
							limit = {
								is_in_array = {
								array = NOE.hololive_union
								value = THIS.id
						}
					country_event = {
						id = hololive_union.1	#event given to leader who can deny or approve the applicant
						days = 1
					}
				}
			  }
		
		
		fire_only_once = yes
	    available = {
				capital_scope = {
					is_on_continent = europe
				}
				has_opinion = {
					target = {
					every_country = {
						limit = {
							has_country_flag = hololive_union_leader
						 }
					 }
					}
					value > 50
				   }
				 NOT = {
				 is_in_array = {
					array = NOE.hololive_union
					value = THIS.id
				 }
		}
	
	}
	visible = {
	NOT = {
				 is_in_array = {
					array = NOE.hololive_union
					value = THIS.id
				 }
		}
	}
}
		leave_hololive_union = {
				icon = GFX_decision_icon_default
				
			  complete_effect = {
				every_country = {
						limit = {
							is_in_array = NOE.hololive_union
						}
					country_event = {
						id = hololive_union.2	#event given to all members of the HU to inform them that a member has left
						days = 1
					}
				}
				remove_from_array = { NOE.hololive_union = ROOT.id 	 }
			  }
		
		
		fire_only_once = yes
	    available = {
				capital_scope = {
					is_on_continent = europe
				}
			is_in_array = {
				array = NOE.hololive_union
				value = THIS.id
			}
			has_capitulated = no
			
		}
}
