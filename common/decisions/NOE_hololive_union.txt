noe_hololive_union = {
				join_hololive_union = {
				icon = GFX_decision_icon_default
		
		complete_effect = {
				every_country = {
							limit = {
								has_country_flag = hololive_union_leader
							}
							country_event = {
								id = hololive_union.1	#event given to leader who can deny or approve the applicant
								days = 1
							}
				}
					set_country_flag = {
						flag = joined_hololive_union
						days = 365
						value = 1
					}
				
					set_temp_variable = {  pupp = THIS.id  }

					every_country = {
					   limit = {
						  is_puppet_of = pupp
					   }
					   add_to_array = { 
							array = NOE.hololive_union 
							value = THIS.id			
						}
					}
		}
		
		
	    available = {
				capital_scope = {
					is_on_continent = europe
				}
				has_opinion = {
					target = NOE
					value > 49
					}
				 NOT = {
				 is_in_array = {
					array = NOE.hololive_union
					value = THIS.id
				 }
				 is_puppet = yes
				}
	
		}
	visible = {
			NOT = {
				 is_in_array = {
					array = NOE.hololive_union
					value = THIS.id
				 }
				has_country_flag = left_hololive_union
		}
	}
}
		leave_hololive_union = {
				icon = GFX_decision_icon_default
				
			  complete_effect = {
				every_country = {
						limit = {
							is_in_array = {
								array = NOE.hololive_union 
								value = THIS.id
							
							}
						}
					country_event = {
						id = hololive_union.2	#event given to all members of the HU to inform them that a member has left
						days = 1
					}
					
				}
				remove_from_array = { NOE.hololive_union = ROOT.id }
				set_country_flag = {
					flag = left_hololive_union
					days = 180
					value = 1
				}
				set_temp_variable = {  home = THIS.id  }

					every_country = {
					   limit = {
						  is_puppet_of = home
					   }
					   remove_from_array = { 
							array = NOE.hololive_union 
							value = THIS.id			
						}
					}
			  }
			  
	    available = {
				capital_scope = {
					is_on_continent = europe
				}
			is_in_array = {
				array = NOE.hololive_union
				value = ROOT.id
			}
			has_capitulated = no
				NOT = {
			has_country_flag = hololive_union_leader
				is_puppet = yes
			}
			
		}
		visible = {
		NOT = {
			has_country_flag = hololive_union_leader
			}
			is_in_array = {
					array = NOE.hololive_union 
					value = THIS.id
			}
			has_country_flag = joined_hololive_union
		}
}
