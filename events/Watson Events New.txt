#Northern ireland solution
country_event = {
	id = watson.14
	title = watson.14.t
	desc = watson.14.d
	picture = GFX_King_George_V
	
	is_triggered_only = yes
	
	option = { #Northern ireland shall be a puppet
		name = watson.14.a
		release_puppet = ULS
	}
	option = { #Northern ireland shall be free
		name = watson.14.b
		release = ULS
	}
	option = { #Northern ireland shall be a returned to ireland
		name = watson.14.c
		trigger = {
			OR = {
				AND = {
					country_exists = FAU
					NOT = {
						country_exists = IRE
					}
				}
				AND = {
					country_exists = IRE
					NOT = {
						country_exists = FAU
					}
				}
			}
		}
				if = {
			limit = {
				country_exists = FAU
			}   
				119 = {
					transfer_state_to = FAU
				}
		}
		else_if = {
			limit = {
				country_exists = IRE
			}
			119 = {
					transfer_state_to = IRE
				}
		}
	}
}
