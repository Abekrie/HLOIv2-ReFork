			add_to_faction = FLA
			add_to_faction = RUS
		}
		if = {
			limit = {
				KRO = {
					has_country_flag = voted_for_black_alliance_kro
				}
			}
			FBK = {
				add_to_faction = KRO
			}
		}
	}
}
