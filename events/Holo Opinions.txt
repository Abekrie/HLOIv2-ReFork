namespace = holo_opinion

#Setting country tags
country_event = {
	id = holo_opinion.1
	hidden = yes
	
	is_triggered_only = yes
	
	immediate = {
		every_other_country = {
			if = {
			limit = {
				has_government = democratic
					OR = {
					has_government = pro_yagoo_progressive
					}
				}
				set_country_flag = yagoo_democratic
			}
			else_if = {
			limit = {
				has_government = pro_yagoo_monarchy
					OR = {
					has_government = pro_yagoo_kleptocracy
					}
				set_country_flag = yagoo_authoritarian
				}
			}
			else_if = {
			limit = {
				has_government = communist
					OR = {
					has_government = pro_resistance_nationalism
					}
					set_country_flag = resistance_revolutionary
					}
			}
			else_if = {
			limit = {
				has_government = pro_resistance_kleptocracy
				}
				set_country_flag = resistance_kleptocracy
			}
			else_if = {
			limit = {
				has_government = pro_resistance_monarchy
				}
				set_country_flag = resistance_monarchy
			}
			else_if = {
			limit = {
				has_government = left_wing_revolutionary
				OR = {
				has_government = fascism
				}
				set_country_flag = anti_revolutionary
				}
			}
			else_if = {
			limit = {
				has_government = democratic_revitalisation
					OR = {
					has_government = reactionary_monarchy
					}
				}
				set_country_flag = anti_reactionary
			}
			else_if = {
			limit = {
				has_government = neutrality
				OR = {
					has_government = reactionary_nationalism
					has_government = divine_mandate
					has_government = pirate_republic
				}
				set_country_flag = neutral_authoritarian
				}
			}
			else_if = {
			limit = {
				has_government = mixed_ideology_democracy
			}
				set_country_flag = neutral_democratic
			}
		}
		country_event = holo_opinion.2
	}
}
#Startup
