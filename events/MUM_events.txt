country_event = {
	id = mumei.5
	title = mumei.5.t
	desc = mumei.5.desc
	picture = GFX_report_event_001
	is_triggered_only = yes
	#Imperialism is wrong, Sudan should be free 
	option = {
		name = mumei.5.a
		add_to_variable = {
			var = MUM.pekora_willing_negotiate
			value = 2
		}
		add_to_variable = {
			var = MUM.yagoo_willing_negotiate
			value = 2
		}
	}
	#We need to restore the pre-war borders
	option = {
		name = mumei.5.b
		add_to_variable = {
			var = MUM.pekora_willing_negotiate
			value = -2
		}
		add_to_variable = {

			value = 3
		}
	}

		}
	}
