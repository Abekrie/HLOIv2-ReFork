				}
			}
		}

				var = MUM.egypt_treaty_acceptance

			}
		}
		if= {
			limit = {
				tag = MUM
			}

		}
		if= {
			limit = {
				tag = PEK

	#disagree

	id = mumegypt.6
	title = mumegypt.6.t
	desc = mumegypt.6.d
	picture = GFX_report_event_MUM_mumei_found
	trigger = {
		AND = {
			OR = {
				tag = PEK
				tag = MUM
				tag = YAG
			}
			check_variable = {
				var = MUM.egypt_treaty_acceptance
				value = 3
				compare = equals
			}
		}
	}
	mean_time_to_happen = {
		days = 15
	}
	#A great success 
	option = {
		name = mumegypt.6.a
		trigger = {
			tag = MUM
		}
		MUM = {
			set_country_flag = MUM_egypt_peace_treaty
		}
		if = {
			limit = {
				MUM = {
					has_country_flag = MUM_korone_devised_treaty
				 }
			}
			if = {
				limit = {
					MUM = {
						has_country_flag = MUM_korone_treaty_steward
					}
				}
				PYW_KRN_steward = yes
			}
			if = {
				limit = {
					MUM = {
						has_country_flag = MUM_korone_treaty_threaten_with_war
					}
				}

				}

			}
		}
		else_if = {
			limit = {

			}
			if = {
				limit = {
