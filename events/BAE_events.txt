#
#	El Estado Ratismo
#
country_event = {
	id = baelz.8
	title = baelz.8.t
	desc = baelz.8.d
	picture = GFX_report_event_BAE_el_estado_ratismo
	is_triggered_only = yes
	#Una ideologia de los colombianos para los colombianos
	option = {
		name = baelz.8.a
		trigger = {
			check_variable = {
				var = BAE.ratismo_support
				value = 8
				compare = greater_than_or_equals
			}
		}
		set_country_flag = BAE_ratismo_generalissimo_flag
		set_politics = {
			ruling_party = divine_mandate
			elections_allowed = no
		}
		add_stability = -0.1
		add_war_support = -0.05
		add_popularity = {
			ideology = divine_mandate
			popularity = 0.35
		}
		add_to_variable = {
			var = BAE.anti_aggression
			value = 0.25
			tooltip = GEN_change_in_anti_aggression
		}
		swap_ideas = {
			remove_idea = BAE_ratismo
			add_idea = BAE_ratismo2
		}
		remove_power_balance = {
			id = BAE_party_balance
		}
		start_civil_war = {
			ruling_party = divine_mandate
			ideology = democratic_revitalisation
			size = 0.2
			keep_all_characters = yes
		}
	}
	#A Counter-Revolution
	option = {
		name = baelz.8.b
		trigger = {
			check_variable = {
				var = BAE.ratismo_support
				value = 8
				compare = less_than
			}
			check_variable = {
				var = BAE.ratismo_support
				value = 4
				compare = greater_than_or_equals
			}
		}
		set_country_flag = BAE_ratismo_generalissimo_flag
		set_politics = {
			ruling_party = divine_mandate
			elections_allowed = no
		}
		add_stability = -0.15
		add_war_support = -0.15
		add_to_variable = {
			var = BAE.anti_aggression
			value = 0.45
			tooltip = GEN_change_in_anti_aggression
		}
		swap_ideas = {
			remove_idea = BAE_ratismo
			add_idea = BAE_ratismo2
		}
		remove_power_balance = {
			id = BAE_party_balance
		}
		start_civil_war = {
			ruling_party = divine_mandate
			ideology = democratic_revitalisation
			size = 0.6
			keep_all_characters = yes
		}
	}
	#Civil war epic fail edition
	option = {
		name = baelz.8.c
		trigger = {
			check_variable = {
				var = BAE.ratismo_support
				value = 4
				compare = less_than
			}
		}
		set_country_flag = BAE_ratismo_generalissimo_flag
		set_politics = {
			ruling_party = divine_mandate
			elections_allowed = no
		}
		add_stability = -0.15
		add_war_support = -0.15
		add_to_variable = {
			var = BAE.anti_aggression
			value = 0.45
			tooltip = GEN_change_in_anti_aggression
		}
		swap_ideas = {
			remove_idea = BAE_ratismo
			add_idea = BAE_ratismo2
		}
		remove_power_balance = {
			id = BAE_party_balance
		}

			ruling_party = divine_mandate
